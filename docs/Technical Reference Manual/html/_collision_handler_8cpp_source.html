80 void CollisionHandler::CheckSegmentMushroomCollisions()
81 {
82  for(unsigned int i = 0; i < centipede_->GetCentipede().size(); i++)
83  {
84  // if segment is poisoned, or moving up/down, don't check for collisions, since the behaviour for poisoned

87  centipede_->GetCentipede().at(i).GetDirection() == Direction::UP ||
88  centipede_->GetCentipede().at(i).GetDirection() == Direction::DOWN ||
89  centipede_->GetCentipede().at(i).IsDead())
90  {
91  continue;
92  }
93 
94  auto it = std::find(field_->GetMushrooms().begin(), field_->GetMushrooms().end(),
95  std::make_pair(centipede_->GetCentipede().at(i).GetRegion(),
96  centipede_->GetCentipede().at(i).GetSubRegion()));
97  while(it != field_->GetMushrooms().end())
98  {
99  if(it->IsPoisoned() && CheckDistanceBetweenEntities(centipede_->GetCentipede().at(i), *it) <
100  CENTIPEDE_SEGMENT_HIT_RADIUS + MUSHROOM_HIT_RADIUS)
101  {
102  // set segments
103  centipede_->GetCentipede().at(i).SetPoisoned(true);
104  // set mushrooms normal after collisions with the last segment of a centipede segment
105  if(centipede_->GetCentipede().at(i).IsLastSegment())
106  {
107  it->SetPoisoned(false);
108  }
109  it++;
110  it = std::find(it, field_->GetMushrooms().end(),
111  std::make_pair(centipede_->GetCentipede().at(i).GetRegion(),
112  centipede_->GetCentipede().at(i).GetSubRegion()));
113  continue;
114  }
115 
116  if(CheckDistanceBetweenEntities(centipede_->GetCentipede().at(i), *it) <
117  CENTIPEDE_SEGMENT_HIT_RADIUS + MUSHROOM_HIT_RADIUS)
118  {
119  if(centipede_->GetCentipede().at(i).GetDirection() == Direction::LEFT)
120  {
121  centipede_->GetCentipede().at(i).SetTopLeftXPosition(it->GetTopLeftXPosition() +
122  MUSHROOM_SPRITE_SIZE);

126  CENTIPEDE_SPRITE_SIDE_SIZE);
127 
128  if(centipede_->GetCentipede().at(i).GetTrajectory() == Trajectory::DOWNWARD)
129  {
130  centipede_->GetCentipede().at(i).SetDirection(Direction::DOWN);
131  }
132 
133  else

135  }
136  it++;
