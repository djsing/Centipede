120 {
121  // find dead segments
122  auto it = std::find(centipede_->GetCentipede().begin(), centipede_->GetCentipede().end() - 1, true);
123  while(it != centipede_->GetCentipede().end() - 1)
124  {
125  if(it != centipede_->GetCentipede().begin())
126  {
127  // Set segment before dead segment to be the last segment of the preceeding new Centipede section
128  it--;
129  it->SetLastSegment(true);

131  // return to found iterator
132  it++;
133  // Set segment after dead segment to be the head segment of the successive new Centipede section
134  it++;
135  it->SetFirstSegment(true);
136  it = std::find(it, centipede_->GetCentipede().end() - 1, true);
137  }
138 
139  // delete collided segments

143  centipede_->GetCentipede().erase(deadBegin, centipede_->GetCentipede().end());
144  }
